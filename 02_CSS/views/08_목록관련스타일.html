순서가 있는 목록의 스타일 지정하기

            선택자 { list-style-type: 
                   decimal // 기본순서
                   decimal-leading-zero // 앞에 0이 붙는 십진수
                   lower-roman // 로마 소문자 표기로 순서를 나타낸다.
                   upper-roman // 로마 대문자 표기로
                   lower-alpha // 영 소문자 표기로
                   upper-alpha // 영 대문자 표기로
            }
